def calculate_building_shadow(building_geom, height_ft, altitude_deg, azimuth_deg):
    """
    Calculate shadow polygon for a building.
    
    Parameters:
    - building_geom: Building footprint geometry
    - height_ft: Building height in feet
    - altitude_deg: Solar altitude angle in degrees
    - azimuth_deg: Solar azimuth angle in degrees (0=North, 90=East)
    
    Returns:
    - Shadow polygon
    """
    # If sun is below horizon or building has no height, no shadow
    if altitude_deg <= 0 or height_ft <= 0:
        return None
    
    # Calculate shadow length
    altitude_rad = np.radians(altitude_deg)
    shadow_length = height_ft / np.tan(altitude_rad)
    
    # Calculate shadow direction (opposite of sun)
    shadow_azimuth = (azimuth_deg + 180) % 360
    shadow_azimuth_rad = np.radians(shadow_azimuth)
    
    # Calculate shadow offset
    dx = shadow_length * np.sin(shadow_azimuth_rad)
    dy = shadow_length * np.cos(shadow_azimuth_rad)
    
    # Create shadow polygon by translating building footprint
    try:
        from shapely.affinity import translate
        shadow = translate(building_geom, xoff=dx, yoff=dy)
        
        # Union with building footprint for full shadow
        full_shadow = unary_union([building_geom, shadow])
        
        return full_shadow.convex_hull if full_shadow.is_valid else None
    except:
        return None

# Helper: compute coverage in a small window quickly
def compute_coverage_for_buffer(buffer_geom, raster_arr, transform, nodata=0):
    """
    Returns fraction of buffer area covered by raster==1.
    Uses a window read from the full raster array + rasterizes the buffer to a mask in that window.
    """
    b = buffer_geom.bounds
    w = from_bounds(*b, transform=transform)

    # Clip window to raster bounds
    row_off = max(0, int(np.floor(w.row_off)))
    col_off = max(0, int(np.floor(w.col_off)))
    row_end = min(raster_arr.shape[0], int(np.ceil(w.row_off + w.height)))
    col_end = min(raster_arr.shape[1], int(np.ceil(w.col_off + w.width)))

    if row_end <= row_off or col_end <= col_off:
        return 0.0

    window_arr = raster_arr[row_off:row_end, col_off:col_end]

    # Build a window-specific transform
    win_transform = rasterio.Affine(
        transform.a, transform.b, transform.c + col_off * transform.a,
        transform.d, transform.e, transform.f + row_off * transform.e
    )

    # Rasterize buffer into this window to create a mask
    mask = rasterize(
        [(mapping(buffer_geom), 1)],
        out_shape=window_arr.shape,
        transform=win_transform,
        fill=0,
        dtype=np.uint8,
        all_touched=True  # keeps behavior closer to pixel-box method
    )

    inside = (mask == 1)
    denom = inside.sum()
    if denom == 0:
        return 0.0

    shaded = (window_arr == 1) & inside
    return shaded.sum() / denom

# Step 3: Extract nodes
print("\n3. Extracting unique network nodes...")
all_coords = set()
for idx, row in edges_wgs84.iterrows():
    coords = list(row.geometry.coords)
    all_coords.add(coords[0])
    all_coords.add(coords[-1])

nodes = []
node_map = {}
for i, (lon, lat) in enumerate(sorted(all_coords)):
    nodes.append({'id': i, 'lon': float(lon), 'lat': float(lat)})
    node_map[(lon, lat)] = i

print(f"   ✓ Extracted {len(nodes):,} nodes")

# Step 4: Prepare edges with shade data
print("\n4. Preparing edges with shade scores...")
edges_data = []
for idx, row in edges_wgs84.iterrows():
    coords = list(row.geometry.coords)
    
    # Get shade scores for all scenarios
    shade_scores = {}
    for col in edges_wgs84.columns:
        if col.startswith('shade_') and 'shadow' not in col:
            scenario_name = col.replace('shade_', '')
            value = row[col]
            shade_scores[scenario_name] = float(value) if pd.notna(value) else 0.0
    
    # Create edge object
    edges_data.append({
        'id': int(idx),
        'source': int(node_map[coords[0]]),
        'target': int(node_map[coords[-1]]),
        'length': float(row.get('length', row.geometry.length)),
        'geometry': [[float(lon), float(lat)] for lon, lat in coords],
        'shade': shade_scores,
        'name': str(row.get('name', '')) if pd.notna(row.get('name')) else ''
    })

print(f"   ✓ Prepared {len(edges_data):,} edges with {len(shade_scores)} shade scenarios each")
